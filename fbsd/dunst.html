Dunst Notifier

Dunst runs on a fairly default config. I adjusted the position of it, so it matches the edges of my other windows in the upper right corner. Furthermore I altered the colors to match my colorscheme.

~/.config/dunst/dunstrc

1: # See dunst(5) for all configuration options
2:
3: [global]
4: ### Display ###
5:
6: # Which monitor should the notifications be displayed on.
7: monitor = 1
8:
9: # Display notification on focused monitor. Possible: mouse, keyboard, none
10: follow = none
11:
12: # constant width of 300
13: width = 300
14:
15: # maximum height of a single notification, excluding the frame
16: height = (0, 300)
17:
18: # Position the notification in the top right corner
19: origin = top-right
20:
21: # Offset from origin (to match my borders from other windows)
22: offset = 50x50
23:
24: # Scale factor. Auto-detecion if 0.
25: scale = 0
26:
27: # Mximum number of notification (0 = no limit)
28: notification_limit = 0
29:
30: ### Progress bar
31:
32: # Turn on progress bar
33: progress_bar = true
34:
35: # set progress bar height including frame
36: progress_bar_height = 10
37:
38: # set frame width of progress bar
39: progress_bar_frame_width = 1
40:
41: # set minimum width of progress bar
42: progress_bar_min_width = 150
43:
44: # set maximum width of progress bar
45: progress_bar_max_width = 300
46:
47: # corner radius of progress bar
48: progress_bar_corner_radius = 0
49:
50: # define which corners to round when drawing progress bar
51: # will be ignored, when progress_bar_corner_radius = 0
52: progress_bar_corners = all
53:
54: # corner radius for the icon image
55: icon_corner_radius = 0
56:
57: # define which corners to round when drawing icon image
58: # will be ignored, when icon_corner_radius = 0
59: icon_corners = all
60:
61: # show how many messages are hurrently hidden (because of notification_limit
62: indicate_hidden = yes
63:
64: # transparency of window, X11 only (we're using wayland/hyprland)
65: transparency = 0
66:
67: # draw a line of given height between two notifications
68: # gets ignored if gap_size > 0
69: separator_height = 2
70:
71: # padding between text and separator
72: padding = 8
73:
74: # Horizontal padding
75: horizontal_padding = 0
76:
77: # padding between text and icon
78: text_icon_padding = 5
79:
80: # set frame thickness
81: frame_width = 3
82:
83: # set frame color
84: frame_color = "#2b7186"
85:
86: # gap size between notifications
87: gap_size = 0
88:
89: # set separator color
90: separator_color = frame
91:
92: # sort type
93: sort = yes
94:
95: ### Text
96:
97: # set font
98: font = GohuFont 14 Nerd Font Mono 11
99:
100: # spacing between lines
101: line_height = 0
102:
103: # allowing html style markups in notifications
104: markup = full
105:
106: # format notifications
107: format = "<b>%s</b>\n%b"
108:
109: # alignment of message text. possible: left, center, right
110: alignment = left
111:
112: # vertical alignment of message text and icon
113: vertical_alignment = center
114:
115: # show age of message if message is older than show_age_threshold seconds.
116: show_age_treshold = 60
117:
118: # specify where to make an ellipsis in long lines.
119: # possible: start, middle, end
120: ellipsize = middle
121:
122: # ignore newlines '\n' in notifications
123: ignore_newline = no
124:
125: # stack together notifications with the same content
126: stack_duplicates = true
127:
128: # hide the count of stacked notifications with the same content
129: hide_uplicate_count = false
130:
131: # display indicators for URLs (u) and actions (A)
132: show_indicators = yes
133:
134: ### Icons
135:
136: # recursive icon lookup. set a single theme instead of having to define look paths
137: enable_recursive_icon_lookup = true
138:
139: # set icon theme (only used for recursive icon lookup
140: icon_theme = Adwaita
141:
142: # align icons, possible: left, right, top, off
143: icon_position = left
144:
145: # scale small icons up to this size
146: min_icon_size = 32
147:
148: # scale larger icons down to this size
149: max_icon_size = 128
150:
151: # path to default icons (not necessary cause we're using recursive lookup)
152: icon_path = /usr/share/icons/gnome/16x16/status/:/usr/share/icons/gnome/16x16/devices/
153:
154: ### History
155:
156: # should a notification popped up from history be sticky or timeout
157: # as it would normally do.
158: sticky_history = yes
159:
160: # maximum amount of notifications kept in history
161: history_length = 20
162:
163: ### Misc/Advanced
164:
165: # dmenu path
166: /usr/bin/dmenu -p dunst
167:
168: # browser for opening urls in context menu
169: browser = /usr/bin/xdg-open
170:
171: # always run rule-defined scrupts, eve nif the notification is suppressed
172: always_run_script = true
173:
174: # define the title of the windows spawned by dunst (X11 only)
175: title = Dunst
176:
177: # define the class of the windows spawned by dunst (X11 only)
178: class = Dunst
179:
180: # radius of corners in pixels
181: corner_radius = 15
